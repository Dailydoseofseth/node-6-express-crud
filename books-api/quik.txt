// import node modules
import express from "express";
import fs from "fs/promises";

// app instance of express
const app = express();

// port
const port = 3000;

// middleware for using JSON
app.use(express.json());

// start server!
app.listen(port, () => {
  console.log(`it's going on port: ${port}`);
});

// ---------------------------------
// Helper Functions
// ---------------------------------

// 1. getAllBooks()
async function getAllBooks() {
  const data = await fs.readFile("books-data.json", "utf8");
  const parsedData = JSON.parse(data);
  return parsedData;
}

// 2. getOneBook(index)
async function getOneBook(index) {
  const data = await fs.readFile("books-data.json", "utf8");
  const parsedData = JSON.parse(data);
  const book = parsedData[index];
  return book;
}

// 1. GET /get-all-books
app.get("/get-all-books", async (req, res) => {
  // call the helper function, store in variable
  const books = await getAllBooks();
  // send json data back as a response
  res.json(books);
});

// 2. GET /get-one-book/:index
app.get("/get-one-book/:index", async (req, res) => {
  // store the index into a variable
  const bookIndex = Number(req.params.index);
  // call the helper function, store in variable

  // send json data back as a response
  res.json(book);

// Create an async function called getOneRecipe
// This function's job is to find and return ONE recipe from our JSON file
async function getOneRecipe(index) {

    // Read the recipes-data.json file
    // await pauses the function until the file has finished loading
    // The file contents are returned as a string because of "utf8"
    const data = await fs.readFile("recipes-data.json", "utf8");

    // Convert the JSON string into a JavaScript array/object
    // This allows us to work with the data using JavaScript
    const parsedData = JSON.parse(data);

    // Get one recipe from the array using the index number provided
    // Example: index 0 gets the first recipe
    const recipe = parsedData[index];

    // Send the selected recipe back to wherever this function was called
    return recipe;
}

// Create a GET endpoint called /get-one-recipe/:index
// The :index is a route parameter that the user supplies in the URL
app.get("/get-one-recipe/:index", async (req, res) => {

    // Get the index value from the URL
    // Convert it from a string into a number
    // Example: "2" becomes 2
    const recipeIndex = Number(req.params.index);

    // Call the getOneRecipe function
    // Pass in the recipe index from the URL
    // Wait for the recipe data to be returned
    const recipe = await getOneRecipe(recipeIndex);

    // Send the recipe back to the client as JSON
    res.json(recipe);
});
